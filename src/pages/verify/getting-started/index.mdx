import Layout from '../../../layouts/mdx';
export default Layout;

import {isIndexPage} from '../../../utils';

import { graphql as gatsbyGraphql, Link } from "gatsby";
export const pageQuery = gatsbyGraphql`
  query GettingStarted {
    pages: allMdx(
      filter: {
        frontmatter: {
          product: { eq: "verify" }
          category: { eq: "Getting Started" }
        }
      }
      sort: { fields: [frontmatter___sort], order: ASC }
    ) {
      edges {
        node {
          __typename
          id
          slug
          fileAbsolutePath
          frontmatter {
            title
            category
            subtitle
          }
        }
      }
    }
  }
`;

Below you will find a list of topics to get you started with Criipto Verify:

<ul>
  {props.data.pages.edges.map(edge => edge.node).filter(node => !isIndexPage(node)).map(node => (
    <li key={node.id}>
      <Link to={`/${node.slug}/`}>{node.frontmatter.title}</Link><br />
      {node.frontmatter.subtitle}
    </li>
  ))}
</ul>
