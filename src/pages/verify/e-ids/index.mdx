import { graphql as gatsbyGraphql, Link } from "gatsby";
export const pageQuery = gatsbyGraphql`
  query eIDs {
    pages: allMdx(
      filter: {
        frontmatter: {
          product: { eq: "verify" }
          category: { eq: "e-IDs" }
        }
      }
      sort: { fields: [frontmatter___sort], order: ASC }
    ) {
      edges {
        node {
          __typename
          id
          slug
          frontmatter {
            title
            category
          }
        }
      }
    }
  }
`;
